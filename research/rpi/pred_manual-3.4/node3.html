Linking

  Note: Regardless of which library (see Library Instances ) is used, -lpred* should be placed BEFORE -lrefdref* in the link line.

Example Link Line 

LIBTYPE=ibm_4
SCSOFT_HOME = /dept/scorec/software
COM_VER            = 2.0
MDB_VER            = 3.4
MEM_VER            = 1.0
MESHOPT_VER        = 3.0
MESHOPTPART_VER    = 2.0
MESHTOOLS_VER      = 3.1
MODEL_VER          = 0.8
PARA_VER           = 8.0
PARA_MODEL_VER     = 0.8
PMDB_VER           = 5.1
PRED_VER           = 3.4
PRED_EXT           = -time
# Hugues' new "old" library
REFDREF_VER        = 1.0
REFDREF_HOME = /dept/scorec/scratch1/decouh2/develop/meshing/refdref
UTIL_VER           = 0.1
LDLIBS   =  \
  -L$(SCSOFT_HOME)/parallel/pred/$(PRED_VER)/lib/$(LIBTYPE) \
  -lpred$(PRED_EXT)-mpi  \
  -L$(REFDREF_HOME)/$(REFDREF_VER)/lib/$(LIBTYPE) \
  -lrefdref-mpi  \
  -L$(SCSOFT_HOME)/parallel/pmdb/$(PMDB_VER)/lib/$(LIBTYPE) \
  -lpmdb-mpi   \
  -L$(SCSOFT_HOME)/meshing/meshTools/$(MESHTOOLS_VER)/lib/$(LIBTYPE) \
  -lmeshTools-mpi  \
  -L$(SCSOFT_HOME)/parasolid/$(PARA_VER) \
  -lfrustrum -lfg -lparasolid \
  -L$(SCSOFT_HOME)/parallel/comm/$(COM_VER)/lib/$(LIBTYPE) \
  -lcomm-mpi  \
  -L$(SCSOFT_HOME)/model/model/$(MODEL_VER)/lib/$(LIBTYPE) \
  -lmodel \
  -L$(SCSOFT_HOME)/model/parasolidModel/$(PARA_MODEL_VER)/lib/$(LIBTYPE) \
  -lparasolidModel \
  -L$(SCSOFT_HOME)/model/mesh/$(MDB_VER)/lib/$(LIBTYPE) \
  -lmesh  \
  -L$(SCSOFT_HOME)/meshing/meshOpt/$(MESHOPT_VER)/lib/$(LIBTYPE) \
  -lmeshOpt \
  -L$(SCSOFT_HOME)/parallel/meshOptPart/$(MESHOPTPART_VER)/lib/$(LIBTYPE) \
  -lmeshOptPart  \
  -L$(SCSOFT_HOME)/util/memory/$(MEM_VER)/lib/$(LIBTYPE) 
  -lmemory  \
  -L$(SCSOFT_HOME)/util/util/$(UTIL_VER)/lib/$(LIBTYPE) 
  -lutil \
  -lm
