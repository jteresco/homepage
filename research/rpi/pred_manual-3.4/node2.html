Include Files

  To use pred, you need to include the following file: 
pred.h
(located in /dept/scorec/software/pred/3.4/include).

A typical INCLUDES variable for use with pred looks like 

SCSOFT_HOME = /dept/scorec/software
COM_VER            = 2.0
MDB_VER            = 3.4
MESHTOOLS_VER      = 3.1
MODEL_VER          = 0.8
PMDB_VER           = 5.1
PRED_VER           = 3.4
PRED_EXT           = -time
# Hugues' new "old" library
REFDREF_VER        = 1.0
REFDREF_HOME       = /dept/scorec/scratch1/decouh2/develop/meshing/refdref
INCLUDES   =  \
           -I$(SCSOFT_HOME)/parallel/pred/$(PRED_VER)/include  \
           -I$(REFDREF_HOME)/$(REFDREF_VER)/include \
           -I$(SCSOFT_HOME)/refdref/$(REFDREF_VER)/include  \
           -I$(SCSOFT_HOME)/parallel/pmdb/$(PMDB_VER)/include    \
           -I$(SCSOFT_HOME)/meshing/meshTools/$(MESHTOOLS_VER)/include  \
           -I$(SCSOFT_HOME)/parallel/comm/$(COM_VER)/include  \
           -I$(SCSOFT_HOME)/model/model/$(MODEL_VER)/include \
           -I$(SCSOFT_HOME)/model/mesh/$(MDB_VER)/include
